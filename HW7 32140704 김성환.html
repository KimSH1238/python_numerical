import numpy as np

def f(x):
    eq=x**2. + np.sin(x) +.1
    #.1로 쓰면 쓰기는 편하지만 가독성이 안좋다
    return eq
def df(x):
    deq=2.*x + np.cos(x)
    return deq

maxN=100
xo=0
ans=np.array([f(xo)])
er=np.array([])

for k in range(maxN):
    xnew=xo-f(xo)/df(xo)
    print(f'N = {k+1} , x = {xnew:0.6f}')

    plus=np.array([xo])
